create or replace TABLE RESTAURANT_DETAILS (
	RESTAURANT_ID NUMBER(10,0) NOT NULL,
	RESTAURANT_CUISINE VARCHAR(1000),
	SHOW_RATE_US BOOLEAN,
	RATING NUMBER(1,0),
